📝MessageGraph 定义

from langgraph.graph
  import MessageGraph

# 专门用于以消息为中心的图
graph = MessageGraph()

# 添加消息节点
graph.add_node("chat", chat_node)
graph.add_edge("start", "chat")
  • 继承自 StateGraph
  • 专门处理消息流
  • 默认使用 add_messages reducer

⚡add_messages 函数

from langgraph.graph.messages
  import add_messages

# 预构建的 Reducer 函数
# - 跟踪消息 ID
# - 追加新消息
