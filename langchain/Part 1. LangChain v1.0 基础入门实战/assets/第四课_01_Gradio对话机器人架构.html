# 1. 初始化模型与系统设定 model = ChatTongyi(model="qwen3-max") system_message = SystemMessage(content="你叫小智") # 2. 定义响应函数（流式输出） def respond(user_msg, chat_hist, messages_list): messages_list.append(HumanMessage(content=user_msg)) chat_hist = chat_hist + [(user_msg, "")] partial = "" for chunk in model.stream(messages_list): partial += chunk.content chat_hist[-1] = (user_msg, partial) yield "", chat_hist, messages_list messages_list.append(AIMessage(content=partial)) yield "", chat_hist, messages_list # 3.
